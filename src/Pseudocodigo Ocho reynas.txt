funcion resolver_reinas(columna):                     → O(1)


    si columna == N:                                  → O(1)
        imprimir solución                             → O(N²)  (imprimir tablero)
        regresar True                                 → O(1)


    para cada fila en tablero:                        → O(N)


        si es_seguro(fila, columna):                  → O(N)
                                                       (revisa fila y diagonales)


            colocar reina                             → O(1)


            si resolver_reinas(columna + 1):          → T(N-1)  (llamada recursiva)


                regresar True                         → O(1)


            quitar reina (backtracking)               → O(1)


    regresar False                                    → O(1)
